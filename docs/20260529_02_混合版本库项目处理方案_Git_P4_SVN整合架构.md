### 3.2 多版本库协调管理器

```python
class MultiVCSManager:
    """多版本库协调管理器"""
    
    def __init__(self):
        self.vcs_handlers = {
            "git": GitVCS(),
            "p4": PerforceVCS(), 
            "svn": SVNVCS()
        }
        self.project_vcs_map: Dict[str, Dict[str, str]] = {}  # project_id -> {path: vcs_type}
    
    async def scan_project_vcs(self, project_root: str) -> Dict[str, str]:
        """扫描项目中的所有版本库"""
        vcs_map = {}
        
        # 扫描根目录
        for vcs_type, handler in self.vcs_handlers.items():
            if await handler.detect(project_root):
                vcs_map[project_root] = vcs_type
                break
        
        # 深度扫描子目录
        for item in Path(project_root).rglob("*"):
            if item.is_dir():
                for vcs_type, handler in self.vcs_handlers.items():
                    if await handler.detect(str(item)):
                        vcs_map[str(item)] = vcs_type
                        break
        
        return vcs_map
    
    async def sync_all_vcs(self, project_id: str) -> Dict[str, bool]:
        """同步项目中的所有版本库"""
        results = {}
        vcs_map = self.project_vcs_map.get(project_id, {})
        
        for path, vcs_type in vcs_map.items():
            handler = self.vcs_handlers[vcs_type]
            try:
                success = await handler.update(path)
                results[path] = success
            except Exception as e:
                logger.error(f"同步 {vcs_type} 失败: {e}")
                results[path] = False
        
        return results
```

### 4.1 智能检测阶段

```python
async def import_mixed_vcs_project(project_root: str, project_name: str):
    """导入混合版本库项目"""
    
    # 1. 扫描所有版本库
    vcs_manager = MultiVCSManager()
    vcs_map = await vcs_manager.scan_project_vcs(project_root)
    
    # 2. 分析依赖关系
    dep_manager = VCSDependencyManager()
    dependencies = await dep_manager.analyze_dependencies(vcs_map)
    
    # 3. 生成项目配置
    project_config = await _generate_mixed_vcs_config(vcs_map, dependencies)
    
    # 4. 创建项目记录
    project_id = await _create_project_record(project_name, project_config)
    
    # 5. 按依赖顺序初始化版本库
    sync_order = await dep_manager.get_sync_order(dependencies)
    for path in sync_order:
        vcs_type = vcs_map[path]
        handler = vcs_manager.vcs_handlers[vcs_type]
        await handler.checkout(path)
    
    return project_id
```

```python
async def _generate_mixed_vcs_config(vcs_map: Dict[str, str], dependencies: Dict[str, List[str]]) -> Dict:
    """生成混合版本库项目配置"""
    
    config = {
        "vcs_strategy": "mixed",
        "primary_vcs": _determine_primary_vcs(vcs_map),
        "vcs_components": {},
        "sync_strategy": "dependent_first",
        "conflict_resolution": "manual",  # 或 "auto_merge"
    }
    
    for path, vcs_type in vcs_map.items():
        config["vcs_components"][path] = {
            "type": vcs_type,
            "dependencies": dependencies.get(path, []),
            "sync_priority": _calculate_sync_priority(path, vcs_type, dependencies),
            "is_primary": path == config["primary_vcs"]
        }
    
    return config
```

```python
class UnifiedVCSOperations:
    """统一版本库操作接口"""
    
    async def sync(self, project_id: str, target_path: str = None) -> bool:
        """同步项目（全部或指定路径）"""
        if target_path:
            # 同步单个版本库
            vcs_type = self._get_vcs_type(project_id, target_path)
            handler = self.vcs_handlers[vcs_type]
            return await handler.update(target_path)
        else:
            # 同步所有版本库（按依赖顺序）
            return await self._sync_all_with_dependencies(project_id)
    
    async def commit(self, project_id: str, message: str, paths: List[str] = None) -> bool:
        """提交变更"""
        if not paths:
            # 提交所有有变更的版本库
            changed_paths = await self._get_changed_paths(project_id)
            return await self._commit_multiple_vcs(project_id, message, changed_paths)
        else:
            # 提交指定路径
            return await self._commit_specific_paths(project_id, message, paths)
    
    async def get_status(self, project_id: str) -> Dict:
        """获取项目整体状态"""
        status_report = {}
        vcs_map = self.project_vcs_map.get(project_id, {})
        
        for path, vcs_type in vcs_map.items():
            handler = self.vcs_handlers[vcs_type]
            status = await handler.get_status(path)
            status_report[path] = status
        
        return {
            "project_id": project_id,
            "vcs_count": len(vcs_map),
            "status_by_path": status_report,
            "overall_status": self._calculate_overall_status(status_report)
        }
```

```python
class CrossVCSConflictResolver:
    """跨版本库冲突解决器"""
    
    async def detect_cross_vcs_conflicts(self, project_id: str) -> List[Conflict]:
        """检测跨版本库冲突"""
        conflicts = []
        
        # 检测文件移动冲突（Git移动 vs P4移动）
        conflicts.extend(await self._detect_file_move_conflicts(project_id))
        
        # 检测依赖版本冲突
        conflicts.extend(await self._detect_dependency_conflicts(project_id))
        
        # 检测配置冲突
        conflicts.extend(await self._detect_config_conflicts(project_id))
        
        return conflicts
    
    async def auto_resolve_conflicts(self, conflicts: List[Conflict]) -> ResolutionResult:
        """自动解决冲突"""
        resolved = []
        unresolved = []
        
        for conflict in conflicts:
            if conflict.severity == "low" and self._has_auto_resolution(conflict):
                resolution = await self._apply_auto_resolution(conflict)
                resolved.append((conflict, resolution))
            else:
                unresolved.append(conflict)
        
        return ResolutionResult(resolved=resolved, unresolved=unresolved)
```
